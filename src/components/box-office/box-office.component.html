Auto-Select Logic 

selectMovie(movie: any) {
  this.selectedMovie = movie;
  
  // If only 1 show, auto-select it
  if (movie.shows.length === 1) {
    this.selectedShow = movie.shows[0];
  } else {
    this.selectedShow = '';
  }
}

Dynamic Calculation

bookTicket() {
  if (this.ticketCount) {
    this.totalCost = 
      this.selectedMovie.rate * this.ticketCount;
  }
}
