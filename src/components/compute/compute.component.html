calculator(type: string, valor: string | number) {
  if (type === 'action') {
    if (valor === 'c') {
      this.resultado = '';
    } else if (['+', '-', '*', '/', '.'].includes(valor)) {
      this.resultado += valor;
    } else if (valor === '=') {
      try {
        this.resultado = eval(this.resultado);
      } catch {
        this.resultado = 'Error';
      }
    }
  } else {
    this.resultado += valor;
  }
}
