Cost Calculation (TS)

addBookingItem() {
  const rate = this.rentType === 'Per Day' 
    ? car.ratePerDay 
    : car.ratePerHr;

  const cost = rate * this.duration * this.qty;
  this.bookingItems.push({
    carName: car.name,
    rentType: this.rentType,
    duration: this.duration,
    quantity: this.qty,
    cost: cost
  });
}
